🚀 Quick Start

import { createFormify } from 'formease';

const formify = createFormify({
  form: '#myForm',
  validation: {
    rules: {
      email: { required: true },
      password: { required: true, minLength: 8 }
    }
  },
  autoSave: {
    storageKey: 'my-form-data'
  },
  onValidationSuccess: (data) => {
    console.log('Form submitted:', data);
  }
});
