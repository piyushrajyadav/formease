Basic Usage

import { createFormify } from '@piyushrajyadav/formease';

const formify = createFormify({
  form: '#myForm',
  validation: {
    rules: {
      email: { required: true },
      password: { required: true, minLength: 8 }
    }
  },
  autoSave: {
    storageKey: 'my-form-data'
  },
  onValidationSuccess: (data) => {
    console.log('Form submitted:', data);
  }
});

Documentation

API Reference

createFormify(config: FormifyConfig)

Creates and returns a new FormEase instance.

import { createFormify } from '@piyushrajyadav/formease';

const formify = createFormify({
  form: '#myForm',
  validation: { /* validation config */ },
  autoSave: { /* autosave config */ }
});

formify.validate(): ValidationResult

Manually validates the entire form and returns the result.

const result = formify.validate();
if (result.isValid) {
  console.log('Form is valid!');
} else {
  console.log('Errors:', result.errors);
}

formify.getFormData(): FormData

Gets current form data as a plain object.

const data = formify.getFormData();
console.log(data); // { email: 'user@example.com', ... }

formify.reset(): void

Resets the form to its initial state and clears all errors.

formify.reset();

formify.destroy(): void

Removes all event listeners and cleans up the instance.

formify.destroy();

Configuration Options

FormifyConfig

interface FormifyConfig {
  form: string | HTMLFormElement;          // Form selector or element
  validation?: ValidationConfig;            // Validation configuration
  autoSave?: AutoSaveConfig;               // Auto-save configuration
  errorHandling?: ErrorConfig;             // Error display configuration
  onValidationSuccess?: (data: FormData) => void;
  onValidationError?: (errors: ErrorMap) => void;
}

ValidationConfig

interface ValidationConfig {
  rules?: Record<string, ValidationRule>;
  validateOnBlur?: boolean;      // Default: true
  validateOnInput?: boolean;     // Default: false
  validateOnSubmit?: boolean;    // Default: true
}

AutoSaveConfig

interface AutoSaveConfig {
  storageKey?: string;           // Default: 'formease_autosave'
  debounceDelay?: number;        // Default: 500ms
  excludeFields?: string[];      // Default: ['password']
  clearOnSubmit?: boolean;       // Default: true
  onSave?: (data: FormData) => void;
  onRestore?: (data: FormData) => void;
}

Custom Validation Rules

createFormify({
  form: '#myForm',
  validation: {
    rules: {
      username: {
        required: true,
        minLength: 3,
        maxLength: 20,
        pattern: /^[a-zA-Z0-9_]+$/,
        custom: (value) => {
          if (value.includes('admin')) {
            return {
              isValid: false,
              errorMessage: 'Username cannot contain "admin"'
            };
          }
          return { isValid: true };
        }
      }
    }
  }
});

Framework Integration

React

import { useEffect, useRef } from 'react';
import { createFormify } from '@piyushrajyadav/formease';

function ContactForm() {
  const formRef = useRef(null);
  const formifyRef = useRef(null);

  useEffect(() => {
    if (formRef.current) {
      formifyRef.current = createFormify({
        form: formRef.current,
        validation: {
          rules: {
            email: { required: true },
            message: { required: true }
          }
        }
      });
    }

    return () => formifyRef.current?.destroy();
  }, []);

  return <form ref={formRef}>{/* form content */}</form>;
}

<template>

onUnmounted(() => {
  formify?.destroy();
});
</script>
